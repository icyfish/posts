### save html code in JSON

```js
var arr = []
for (var i = 0; i < 256; i++) {
  var char = String.fromCharCode(i)
  if (encodeURI(char) !== encodeURIComponent(char)) {
    arr.push({
      character: char,
      escape: escape(char),
      encodeURI: encodeURI(char),
      encodeURIComponent: encodeURIComponent(char), 
    })
  }
}
console.table(arr)
```
